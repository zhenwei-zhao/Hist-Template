\makeatletter
\renewcommand{\@cftmaketoctitle}{%
  \addpenalty\@secpenalty
  \if@cfthaschapter
    \vspace*{\cftbeforetoctitleskip}%
  \else
    \vspace{\cftbeforetoctitleskip}%
  \fi
  \@cftpagestyle
  {\interlinepenalty\@M
   \centering\hei\bfseries\zihao{-3}\contentsname\par
   \cftmarktoc
   \par\nobreak
   \vskip \cftaftertoctitleskip
